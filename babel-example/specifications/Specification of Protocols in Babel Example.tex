\begin{algorithm}[h!]

\caption{Flood Broadcast}
\begin{tabbing}
xxx\=xxx\=xxx\=xxx\=xxx\=xxx\=xxx\kill
\textbf{Interface:}\\
\> \textbf{Requests:}\\
\> \> \textbf{BroadcastRequest ( $s$, $mid$, $m$ )} \texttt{//s is the sender, m the message}\\
\> \textbf{Indications:}\\
\> \> \textbf{BroadcastDeliver ( $s$, $mid$ $m$ )} \texttt{//s is the sender, m the message}\\
~\\
\textbf{State:}\\
\> neighbors \texttt{set with all neighbors}\\
\> received \texttt{set with message identifiers already received}\\
~\\
\> \textbf{Upon Init () do:}\\
\> \> neighbors $\longleftarrow \{ \}$\\
\> \> received  $\longleftarrow \{ \}$\\
~\\
\> \textbf{Upon BroadcastRequest ( $s$, $mid$, $m$ ) do:}\\
\> \> \textbf{Call} processFloodMessage( \textsc{FloodMSG}, \{$s$, $mid$, $m$\} )\\
~\\
\> \textbf{Upon Receive ( \textsc{FloodMSG}, \{$s$, $mid$, $m$\} ) do:}\\
\> \> \textbf{Call} processFloodMessage( \{$s$, $mid$, $m$\} )\\
~\\
\> \textbf{Procedure} processFloodMessage ( \{$s$, $mid$, $m$\} )\\
\> \> \textbf{If} $mid$ $\exists$ received \textbf{Then}\\
\> \> \> received $\longleftarrow$ received $\cup$ $mid$\\
\> \> \> \textbf{Trigger} BroadcastDeliver ( $s$, $mid$ $m$ )\\
\> \> \> \textbf{Foreach} $p \in $ neighbors \textbf{do}\\
\> \> \> \> \textbf{Trigger} Send ( $p$, \textsc{FloodMSG}, \{$s$, $mid$, $m$\} )\\
~\\
\> \textbf{Upon NeighborUP} ( $p$ ) \textbf{do}\\
\> \> neighbors $\longleftarrow$ neighbors $\cup$ $\{p\}$\\
~\\
\> \textbf{Upon NeighborDown} ( $p$ ) \textbf{do}\\
\> \> neighbors $\longleftarrow$ neighbors $\setminus$ $\{p\}$\\
~\\
\end{tabbing}
\end{algorithm}
